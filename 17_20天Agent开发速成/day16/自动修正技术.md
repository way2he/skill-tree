## 一、自动修正的核心思想

### 反思回路（Reflection Loop）

```
生成代码 → 执行 → 分析错误 → 反思 → 修正 → 重新执行
                                        ↓
                                  成功？→ 是，结束
                                        ↓ 否
                                  次数超限？→ 是，放弃
                                        ↓ 否
                                  继续迭代
```

### 关键概念

| 概念 | 说明 |
|------|------|
| **反思（Reflection）** | 分析失败原因，理解问题所在 |
| **修正（Fix）** | 根据反思结果修改代码 |
| **迭代（Iteration）** | 多次尝试，直到成功或达到限制 |
| **最大尝试次数** | 防止无限循环，设置上限 |

## 二、自动修正系统设计

### 1. 基础自动修正器

```python
class AutoFixer:
    def __init__(self, code_generator, sandbox, error_analyzer, max_attempts=3):
        self.code_generator = code_generator
        self.sandbox = sandbox
        self.error_analyzer = error_analyzer
        self.max_attempts = max_attempts
        
    def auto_fix(self, requirement, initial_code=None):
        current_code = initial_code or self.code_generator.generate(requirement)
        attempts = 0
        history = []
        
        while attempts &lt; self.max_attempts:
            attempts += 1
            print(f"尝试 {attempts}/{self.max_attempts}...")
            
            # 执行代码
            result = self.sandbox.execute_code(current_code)
            
            # 记录历史
            history.append({
                "attempt": attempts,
                "code": current_code,
                "result": result
            })
            
            # 检查是否成功
            if result["success"]:
                print("✅ 成功！")
                return {
                    "success": True,
                    "final_code": current_code,
                    "result": result,
                    "attempts": attempts,
                    "history": history
                }
            
            # 分析错误
            analysis = self.error_analyzer.analyze(current_code, result.get("output", ""))
            print(f"❌ 失败：{analysis}")
            
            # 修正代码
            current_code = self._fix_code(requirement, current_code, analysis, history)
            
        # 达到最大尝试次数
        print("❌ 达到最大尝试次数，放弃")
        return {
            "success": False,
            "final_code": current_code,
            "attempts": attempts,
            "history": history
        }
        
    def _fix_code(self, requirement, current_code, analysis, history):
        # 简单版本：用LLM修正
        prompt = f"""你是一个专业的Python修复专家。请根据以下信息修正代码。

需求：
{requirement}

当前代码：
```python
{current_code}
```

错误信息：
{analysis}

之前的尝试历史：
{self._format_history(history)}

请直接输出修正后的完整代码，不要其他解释。"""
        
        fixed_code = self.code_generator.generate_with_prompt(prompt)
        return fixed_code
        
    def _format_history(self, history):
        lines = []
        for item in history:
            lines.append(f"尝试 {item['attempt']}:")
            lines.append(f"结果: {'成功' if item['result']['success'] else '失败'}")
            if not item['result']['success']:
                lines.append(f"错误: {item['result'].get('output', '')[-200:]}")
        return "\n".join(lines)
```

### 2. 带反思的高级自动修正器

```python
class ReflectiveAutoFixer(AutoFixer):
    def _fix_code(self, requirement, current_code, analysis, history):
        # 先反思，再修正
        reflection = self._reflect(requirement, current_code, analysis, history)
        
        # 根据反思结果修正
        fixed_code = self._fix_with_reflection(requirement, current_code, reflection, history)
        
        return fixed_code
        
    def _reflect(self, requirement, current_code, analysis, history):
        prompt = f"""你是一个专业的代码调试专家。请分析以下情况，思考问题所在。

需求：
{requirement}

当前代码：
```python
{current_code}
```

错误分析：
{analysis}

之前的尝试：
{len(history)} 次尝试都失败了

请回答以下问题：
1. 这个错误的根本原因是什么？
2. 之前的尝试为什么没有解决问题？
3. 应该从哪个方向修正？
4. 具体需要修改哪些地方？

请详细思考，不要直接写代码。"""
        
        reflection = self.code_generator.llm.generate(prompt)
        return reflection
        
    def _fix_with_reflection(self, requirement, current_code, reflection, history):
        prompt = f"""你是一个专业的Python修复专家。请根据以下信息修正代码。

需求：
{requirement}

当前代码：
```python
{current_code}
```

反思分析：
{reflection}

请直接输出修正后的完整代码。"""
        
        fixed_code = self.code_generator.generate_with_prompt(prompt)
        return fixed_code

## 三、提高修正成功率的技巧

### 1. 渐进式修正

不要尝试一次性修复所有问题，分步骤来：

```python
class IncrementalAutoFixer(ReflectiveAutoFixer):
    def _fix_code(self, requirement, current_code, analysis, history):
        # 分析错误类型
        error_type = analysis.get("error_type")
        
        # 简单错误直接修复
        if error_type in ["SyntaxError", "NameError", "TypeError"]:
            return self._fix_simple_error(requirement, current_code, analysis)
        
        # 复杂错误渐进式修复
        else:
            return self._fix_complex_error(requirement, current_code, analysis, history)
            
    def _fix_simple_error(self, requirement, current_code, analysis):
        # 简单错误：直接用LLM修复
        prompt = f"""修复以下代码中的{analysis['error_type']}。

代码：
```python
{current_code}
```

错误：{analysis['error_message']}

直接输出修复后的完整代码。"""
        
        return self.code_generator.generate_with_prompt(prompt)
        
    def _fix_complex_error(self, requirement, current_code, analysis, history):
        # 复杂错误：先简化问题，再逐步恢复
        prompt = f"""当前代码有问题，让我们分步解决。

需求：
{requirement}

当前代码（有问题）：
```python
{current_code}
```

错误：{analysis['error_message']}

请先写一个最简单的版本，只实现核心功能，能运行起来。然后我们再逐步添加其他功能。

输出最简单的可运行版本。"""
        
        simplified_code = self.code_generator.generate_with_prompt(prompt)
        return simplified_code
```

### 2. 利用历史经验

学习过去成功和失败的案例：

```python
class ExperienceBasedAutoFixer(ReflectiveAutoFixer):
    def __init__(self, code_generator, sandbox, error_analyzer, max_attempts=3):
        super().__init__(code_generator, sandbox, error_analyzer, max_attempts)
        self.success_cases = []
        self.failure_cases = []
        
    def auto_fix(self, requirement, initial_code=None):
        # 先查找相似案例
        similar_case = self._find_similar_case(requirement)
        
        if similar_case:
            print(f"找到相似案例：{similar_case}")
            initial_code = similar_case["code"]
            
        result = super().auto_fix(requirement, initial_code)
        
        # 记录经验
        if result["success"]:
            self.success_cases.append({
                "requirement": requirement,
                "code": result["final_code"]
            })
        else:
            self.failure_cases.append({
                "requirement": requirement,
                "history": result["history"]
            })
            
        return result
        
    def _find_similar_case(self, requirement):
        # 简单实现：用关键词匹配
        for case in self.success_cases:
            # 检查关键词重叠
            req_words = set(requirement.lower().split())
            case_words = set(case["requirement"].lower().split())
            overlap = len(req_words &amp; case_words)
            if overlap &gt;= 3:
                return case
        return None
```

### 3. 测试驱动开发（TDD）风格

先写测试，再写代码：

```python
class TDDGuideFixer(ReflectiveAutoFixer):
    def auto_fix(self, requirement, initial_code=None, tests=None):
        # 如果没有提供测试，先生成测试
        if tests is None:
            tests = self._generate_tests(requirement)
            
        current_code = initial_code or self.code_generator.generate(requirement)
        attempts = 0
        history = []
        
        while attempts &lt; self.max_attempts:
            attempts += 1
            
            # 运行测试
            test_result = self._run_tests(current_code, tests)
            
            history.append({
                "attempt": attempts,
                "code": current_code,
                "test_result": test_result
            })
            
            if test_result["all_passed"]:
                return {
                    "success": True,
                    "final_code": current_code,
                    "tests": tests,
                    "attempts": attempts,
                    "history": history
                }
            
            # 分析测试失败原因
            analysis = self._analyze_test_failure(test_result)
            
            # 修正代码
            current_code = self._fix_for_test(requirement, current_code, tests, analysis)
            
        return {
            "success": False,
            "final_code": current_code,
            "tests": tests,
            "attempts": attempts,
            "history": history
        }
        
    def _generate_tests(self, requirement):
        prompt = f"""根据以下需求，生成5个测试用例。

需求：
{requirement}

请输出pytest格式的测试代码。"""
        return self.code_generator.generate_with_prompt(prompt)
        
    def _run_tests(self, code, tests):
        # 简化实现
        return {"all_passed": False, "failures": []}
        
    def _analyze_test_failure(self, test_result):
        return test_result
        
    def _fix_for_test(self, requirement, current_code, tests, analysis):
        prompt = f"""根据以下测试失败情况修正代码。

需求：
{requirement}

当前代码：
```python
{current_code}
```

测试代码：
```python
{tests}
```

测试失败：{analysis}

请修复代码使其通过所有测试。"""
        return self.code_generator.generate_with_prompt(prompt)
```

## 四、实战示例

### 示例：自动修复一个计算平均值的函数

**需求：**
```
写一个函数，输入一个数字列表，返回平均值。
如果列表为空，返回None。
```

**初始代码（有bug）：**
```python
def calculate_average(numbers):
    total = sum(numbers)
    count = len(numbers)
    return total / count
```

**第1次执行：**
- 错误：ZeroDivisionError（空列表时）
- 反思：需要处理空列表情况
- 修正：添加空列表检查

**第1次修正后的代码：**
```python
def calculate_average(numbers):
    if not numbers:
        return None
    total = sum(numbers)
    count = len(numbers)
    return total / count
```

**第2次执行：**
- 成功！
- 完成
